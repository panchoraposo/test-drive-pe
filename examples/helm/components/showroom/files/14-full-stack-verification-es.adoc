== Checklist de pruebas de API

=== Backend API (API Key)

[.console-input]
[source,bash,subs="attributes+"]
----
export APP_NAME="${APP_NAME:-neuralbank-backend}"
export NS="{user_name}-neuralbank"
export APIKEY_NAME="{user_name}-${APP_NAME}"

SECRET_NAME=$(oc get apikey -n "$NS" "$APIKEY_NAME" -o jsonpath='{.status.secretRef.name}')
API_KEY=$(oc get secret -n "$NS" "$SECRET_NAME" -o jsonpath='{.data.api_key}' | tr -d '\n' | base64 -d)

# Debe retornar 200 con datos de clientes
curl -s -w "\nHTTP: %\{http_code}\n" \
  -H "X-API-Key: $API_KEY" \
  "https://neuralbank-backend-{user_name}-neuralbank.{cluster_domain}/api/customers" \
  | tail -1

# Debe retornar 401 sin API key
curl -s -o /dev/null -w "HTTP: %\{http_code}\n" \
  "https://neuralbank-backend-{user_name}-neuralbank.{cluster_domain}/api/customers"
----

=== Stack pre-desplegado (OIDC)

[.console-input]
[source,bash,subs="attributes+"]
----
TOKEN=$(curl -s -X POST \
  "https://rhbk.{cluster_domain}/realms/neuralbank/protocol/openid-connect/token" \
  -H "Content-Type: application/x-www-form-urlencoded" \
  -d "grant_type=password" \
  -d "client_id=neuralbank-frontend" \
  -d "username={user_name}" \
  -d "password=redhat" \
  | jq -r '.access_token')

# Debe retornar 200
curl -s -w "\nHTTP: %\{http_code}\n" \
  -H "Authorization: Bearer $TOKEN" \
  "https://neuralbank.{cluster_domain}/api/v1/customers" \
  | tail -1
----
